@name Epitrochoid(DEElekgolo)
@outputs X Y
@persist Time
runOnTick(1)
#Increase the Timer for faster line draws
Time+=1
#When editing A,B, and C. Make sure that A,B, and C aren't the same
Scale=0.1
A=6*Scale
B=1*Scale
C=3*Scale

X=((A+B)*cos(Time)) - C*cos((A/B+1)*Time)
Y=((A+B)*sin(Time)) - C*sin((A/B+1)*Time)
