@name Lissajous Curve(DEElekgolo)
@outputs X Y Timer
@persist Timer
runOnTick(1)
#Increase the Timer for faster line draws
Timer+=1
#Edit A and B to change the Lissajous ratio
#Values Must be under 1
#If ratio is 1:1 then a circle will be drawn
Scale=0.1
A=5*Scale
B=4*Scale

X=A*sin(A*Timer*pi())
Y=B*sin(B*Timer*pi())
