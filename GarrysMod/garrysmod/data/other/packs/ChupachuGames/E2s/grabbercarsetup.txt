@name Facer
@inputs Target:vector A D Mouse1 Mouse2
@outputs Fire Fire2
@persist Ang:angle P Y R ZDist Angle
@model model/Gib/HGIB.mdl
interval(10)

Target = owner():aimPos()

E = entity():isWeldedTo()

P = E:elevation(Target)
if (A == 1) {Y = -15} else{Y = 0}
if (D == 1) {Y = 15} else{Y = 0}
R = 0 - E:angles():roll()

Ang = ang(P,Y,0)*100
E:applyForce(-E:vel()*2)
E:applyAngForce((-Ang - $Ang*5) - E:angVel())
Fire = Mouse1
Fire2 = Mouse2
