@name CustomAirDrag
@inputs [WheelFL WheelFR WheelRL WheelRR Engine Trans DiffF DiffR Base]:entity Active
@outputs 
@persist
@trigger 

if(changed(Active==0)&Active==1){

    WheelFL:setCustomAirDrag(1,0,0)
    WheelFR:setCustomAirDrag(1,0,0)
    WheelRL:setCustomAirDrag(1,0,0)
    WheelRR:setCustomAirDrag(1,0,0)
    
    Engine:setCustomAirDrag(1,0,0)
    Trans:setCustomAirDrag(1,0,0)
    DiffF:setCustomAirDrag(1,0,0)
    DiffR:setCustomAirDrag(1,0,0)
    
    Base:setCustomAirDrag(1,0,0)
}
