
float mathFloor(float a) {
    fint a;
    return a;
}

float mathRandom(float a, b) {
    float c;
    rand c;
    c = a + (b - a) * c;
    return c;
}

float mathRandomInt(float a, b) {
    float c;
    rand c;
    c = a + (b - a) * c;
    fint c;
    return c;
}

float mathClamp(float a, b, c) {
    if (a < b) return b
    if (a > c) return c
    return a;
}

float mathMin(float a, b) {
    min a, b;
    return a;
}

float mathMax(float a, b) {
    max a, b;
    return a;
}

float mathRound(float a) {
    frnd a;
    return a;
}

float mathFrac(float a) {
    ffrac a;
    return a;
}

float mathPow(float a, b) {
    fpwr a, b
    return a;
}

float mathRoot(float a, b) {
    b = 1 / b;
    fpwr a, b
    return a;
}
